const express = require('express'); const cors = require('cors'); const OpenAI = require('openai'); const app = express(); const port = 3000; const openai = new OpenAI({ apiKey: 'YOUR_API_KEY', }); app.use(cors()); app.use(express.json()); app.post('/api/chat', async (req, res) => { try { const chatCompletion = await openai.chat.completions.create({ model: "gpt-3.5-turbo", messages: req.body.messages, }); res.json(chatCompletion.choices[0].message); } catch (error) { res.status(500).json({ error: 'Failed to fetch from OpenAI' }); } }); app.listen(port, () => { console.log(`Server is running at http://localhost:${port}`); });
